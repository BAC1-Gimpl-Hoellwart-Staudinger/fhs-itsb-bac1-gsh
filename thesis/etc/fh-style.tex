\ifdefined\withTimesNewRoman
  % Set font to Times New Roman. Note that fontspec requires lualatex or xelatex.
  % Requires ttf-mscorefonts-installer on Debian.
  \usepackage{fontspec}
  \setmainfont{Times New Roman}
\fi

\usepackage{a4wide}

% formating margins and skips
\usepackage[a4paper,left=2.5cm,right=2.5cm,top=2.5cm,bottom=4cm]{geometry}
\usepackage{setspace}
\onehalfspacing
\setlength{\parindent}{0pt}
\setlength{\headheight}{18pt}
\setlength{\parskip}{5pt plus 2pt minus 1pt}

% formatting header and footer
\usepackage{fancyhdr}
\pagestyle{empty} % default
\fancyhf{} % clear all header and footer fields
\renewcommand{\sectionmark}[1]{\markboth{#1}{}} % remove section number from header
\renewcommand{\headrulewidth}{0.4pt}
\newcommand{\customleftmark}[0]{\hspace{-8pt}\leftmark} % fix indent

\newcommand{\pageAuthor}[0]{NAME DES AUTORS}

\fancypagestyle{custom-default}{
	\fancyhf{}
	\fancyhead[L]{\textit{\MakeUppercase{\customleftmark}}}
	\fancyhead[R]{\thepage}
}

\fancypagestyle{roman-caps}{
	\fancyhf{}
	\fancyhead[R]{\Roman{page}}
}

\fancypagestyle{theoretical-part}{
	\fancyhf{}
	\fancyhead[L]{\textit{\MakeUppercase{Theoretischer Teil: \customleftmark}}}
	\fancyhead[C]{\pageAuthor}
	\fancyhead[R]{\thepage}
}

\fancypagestyle{practical-part}{
	\fancyhf{}
	\fancyhead[L]{\textit{\MakeUppercase{Praktischer Teil: \customleftmark}}}
	\fancyhead[C]{\pageAuthor}
	\fancyhead[R]{\thepage}
}

% custom reference (blue and clickable)
\newcommand{\defaultRef}[1]{\textcolor{blue}{\hyperref[#1]{\autoref{#1}}}}
